Webpack的Less-Loader

  1. npm install webpack --save-dev
  2. npm init (设置项目信息)
  3. npm install --save-dev css-loader
  4. npm install --save-dev style-loader
  5. npm install --save-dev less-loader
  6. npm install --save-dev less
  7. 新建webpack.config.js 进行webpack的配置
  8. 配置npm的任务

const path = require('path') module.exports = { entry: 'main.js', output: { path: path.resolve(__dirname, 'dist'), filename: 'bundle.js' }, module: { rules: [{ test: /\.css$/, use: ["style-loader", "css-loader"] }] } }
